\begin{frame}{Conteo doble}
  \[
    D(n) = \{\text{divisores de } n\}
  \]

  \[
    \frac{1}{N} \sum_{n=1}^{N} |D(n)|
  \]

  \[
    \left\{
      (m, n) \in \mathbb{N}^2 \;\middle|\;
      \textcolor{red}{m \mid n}
      \text{ y }
      n \leq N
    \right\}
  \]

  \begin{block}{$m \mid n$}
    \begin{itemize}
      \item ``$m$ divide a $n$''
      \item ``$m$ es divisor de $n$''
      \item ``$n$ es múltiplo de $m$''
      \item $\exists k : n = k m$
    \end{itemize}
  \end{block}
\end{frame}
